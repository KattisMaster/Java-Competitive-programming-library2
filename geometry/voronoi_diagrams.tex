\problemtitle{Voronoi diagrams}

\begin{algorithm}{Simple Delaunay triangulation}
\reflisting{delaunay simple}
\complexity{n^4}
\usage{{\tt delaunay(points p, int n, trifun)}}

\desc
Uses a {\tt trifun(int, int, int)} to return all possible delaunay triangles as tripple indices to the point vector. 
\note{Triangles may overlap if points are cocircular.}
\end{algorithm}

\begin{algorithm}{Convex hull Delaunay triangulation}
\reflisting{delaunay hull}
\complexity{\textrm{3d convex hull}}
\usage{{\tt delaunay(points p, int n, trifun)}}

\desc
Returns an arbitrary triangulation if points are cocircular.
\note{Depending on convex hull implementation it may fail if \emph{all} points
are cocircular, as is currently the case.}
\end{algorithm}
