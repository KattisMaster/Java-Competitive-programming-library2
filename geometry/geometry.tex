\problemtitle{Nearest Neighbour}
\begin{algorithm}{Divide and conquer}
\reflisting{closest pair}
\complexity{n\log n}
\usage{{\tt closestpair( points p, int n, \&i1, \&i2 )}}

\desc
{\tt i1, i2} are the indices to the closest pair of points in the point
vector {\tt p} after the call. The distance is returned.
\end{algorithm}
\begin{algorithm}{Simpler method}
\reflisting{closest pair simple}
\complexity{n^2 \textrm{ (average $n$)}}
\usage{{\tt closestpair( points p, int n, \&i1, \&i2 )}}
\end{algorithm}
